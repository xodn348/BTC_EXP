\section*{Experiment Parameters Summary}

\section*{1. Miner Value Function}

\vspace{3pt}
\begin{equation*}
V_i = \mathbb{E}\left[\sum_{t=0}^{T-1} \gamma^t \, \Pi_i(S_t, a_i)\right]
\end{equation*}

\vspace{4pt}
\textbf{Action selection:}
\begin{equation*}
a_i = \begin{cases} \text{honest} & \text{if } V_i^{\text{hon}} > V_i^{\text{dev}} \\ \text{deviate} & \text{otherwise} \end{cases}
\end{equation*}

\section*{2. State Variables: $S_t = \{F_t, M_t, h_i, \delta_t\}$}

\vspace{3pt}
\textbf{Fee:}
\begin{equation*}
F_t = F_{\text{rate}} \cdot B_{\text{MB}} \cdot vB_{\text{per MB}}
\end{equation*}
\begin{itemize}
  \item $F_{\text{rate}}$: sampled from fee distribution (sat/vB)
  \item $vB_{\text{per MB}} = 609{,}692$ vB/MB (median)
  \item $B_{\text{MB}} \in \{1, 2, 4, 8\}$ MB
\end{itemize}

\vspace{4pt}
\textbf{MEV:}
\begin{equation*}
M_t \sim \text{ZILogNormal}
\end{equation*}
\begin{itemize}
  \item 80\% probability mass at zero
\end{itemize}

\vspace{4pt}
\textbf{Miner share:}
\begin{equation*}
h_i \text{ (fixed)}
\end{equation*}
\begin{itemize}
  \item $N=13$ miners, 99.52\% coverage
\end{itemize}

\vspace{4pt}
\textbf{Network delay:}
\begin{align*}
\delta_{\text{hon}}(B) &= \delta_{\text{base}} + \kappa \cdot B_{\text{MB}} \\
\delta_{\text{dev}}(B) &= \delta_{\text{hon}}(B) + w
\end{align*}
\begin{itemize}
  \item $\delta_{\text{base}} = 742$ ms
  \item $\kappa = 26.40$ ms/MB
  \item $w \in \{0.5, 1.0, 2.0\}$ s
\end{itemize}

\section*{3. Payoff Function: $\Pi_i(S_t, a_i)$}

\vspace{3pt}
\textbf{Honest strategy:}
\begin{equation*}
\Pi_i^{\text{hon}} = p_i^{\text{hon}} \cdot X_t - C_i
\end{equation*}

\vspace{4pt}
\textbf{Deviation strategy:}
\begin{equation*}
\Pi_i^{\text{dev}} = p_i^{\text{dev}} \cdot (X_t + G_t) - C_i
\end{equation*}
\begin{itemize}
  \item where $X_t = F_t + M_t$
\end{itemize}

\section*{4. Success Probability: $p_i$}

\vspace{3pt}
\begin{align*}
p_i^{\text{hon}} &= h_i \cdot (1 - \rho_{\text{hon}}(B)) \\
p_i^{\text{dev}} &= h_i \cdot (1 - \rho_{\text{dev}}(B))
\end{align*}

\vspace{4pt}
\textbf{Orphan rate:}
\begin{align*}
\rho_{\text{hon}}(B) &= 1 - e^{-\lambda \delta_{\text{hon}}(B)} \\
\rho_{\text{dev}}(B) &= 1 - e^{-\lambda \delta_{\text{dev}}(B)}
\end{align*}
\begin{itemize}
  \item $\lambda = 0.00167$ blocks/s (Bitcoin: 1 block/600s)
\end{itemize}

\section*{5. Deviation Gain: $G_t$}

\vspace{3pt}
\begin{equation*}
G_t = \alpha \cdot (F_{t-1} + M_{t-1})
\end{equation*}
\begin{itemize}
  \item $\alpha \in \{0.1, 0.25, 0.5\}$ (currently disabled: $G_t = 0$)
\end{itemize}

\section*{6. Cost: $C_i$}

\vspace{3pt}
\begin{itemize}
  \item From miner\_cost\_curve.parquet (CBECI data, USD/block)
\end{itemize}

\columnbreak

\section*{7. Parameter Values}

\vspace{3pt}
\textbf{Discount factor:}
\begin{itemize}
  \item $\gamma = 0.99$
\end{itemize}

\vspace{4pt}
\textbf{Simulation:}
\begin{itemize}
  \item $T = 10{,}000$ rounds
  \item $\lambda = 0.00167$ blocks/s
\end{itemize}

\vspace{4pt}
\textbf{Block size:}
\begin{itemize}
  \item $B_{\text{MB}} \in \{1, 2, 4, 8\}$ MB
\end{itemize}

\vspace{4pt}
\textbf{Network delay:}
\begin{itemize}
  \item $\delta_{\text{base}} = 742$ ms
  \item $\kappa = 26.40$ ms/MB
  \item $w \in \{0.5, 1.0, 2.0\}$ s
\end{itemize}

\vspace{4pt}
\textbf{Base fee:}
\begin{itemize}
  \item basefee$_0$ = 1.0
  \item fill\_target = 0.5
  \item fill\_scale = 40.0 sat/vB
  \item $\phi \in \{0.06, 0.125, 0.25\}$
\end{itemize}

\vspace{4pt}
\textbf{Other:}
\begin{itemize}
  \item Fork rate threshold: $\rho = 0.05$
  \item Deviation extraction: $\alpha \in \{0.1, 0.25, 0.5\}$
\end{itemize}

\section*{8. Formula Relationships}

\vspace{3pt}
\textbf{1. State $\rightarrow$ Delay:}
\begin{equation*}
\delta_{\text{hon}}(B) = 742 + 26.40 \cdot B_{\text{MB}} \quad \text{(ms)}
\end{equation*}

\vspace{4pt}
\textbf{2. Delay $\rightarrow$ Orphan Rate:}
\begin{equation*}
\rho(B) = 1 - e^{-0.00167 \cdot \delta(B)}
\end{equation*}

\vspace{4pt}
\textbf{3. Orphan Rate $\rightarrow$ Success Prob:}
\begin{equation*}
p_i = h_i \cdot (1 - \rho(B))
\end{equation*}

\vspace{4pt}
\textbf{4. State + Action $\rightarrow$ Payoff:}
\begin{align*}
\Pi_i^{\text{hon}} &= p_i^{\text{hon}} \cdot (F_t + M_t) - C_i \\
\Pi_i^{\text{dev}} &= p_i^{\text{dev}} \cdot (F_t + M_t + G_t) - C_i
\end{align*}

\vspace{4pt}
\textbf{5. Payoff $\rightarrow$ Value:}
\begin{equation*}
V_i = \sum_{t=0}^{T-1} \gamma^t \Pi_i(S_t, a_i)
\end{equation*}

\vspace{4pt}
\textbf{6. Value $\rightarrow$ Action:}
\begin{equation*}
a_i = \arg\max_{a \in \{\text{hon}, \text{dev}\}} V_i^a
\end{equation*}
